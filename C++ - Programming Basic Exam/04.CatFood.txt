#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int cats;
	cin >> cats;

	int firstGroup = 0;
	int secondGroup = 0; 
	int third = 0;

	
	double foodQuantity = 0;
	double price = 0;

	for (int i = 0; i < cats; i++) {
		int food;
		cin >> food;
		if (food >= 100 && food < 200) {
			firstGroup++;
		}
		else if (food >= 200 && food < 300) {
			secondGroup++;
		}
		else if (food >= 300 && food < 400) {
			third++;	
		}
		foodQuantity = foodQuantity + food;
	}
	price = foodQuantity * 12.45;
	double finalPrice = price / 1000;
	cout << "Group 1: " << firstGroup << " cats." << endl;
	cout << "Group 2: " << secondGroup << " cats." << endl;
	cout << "Group 3: " << third << " cats." << endl;
	cout.setf(ios::fixed);
	cout.precision(2);
	cout << "Price for food per day: " << finalPrice << " lv.";


}
