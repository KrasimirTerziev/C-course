#include <iostream>
#include <cmath>
#include <sstream>
#include <climits>
using namespace std;

int main()
{
	int food;
	cin >> food;

	int allFood = 0;
	int grToKg = food * 1000;

	for (int i = 0; i < INT_MAX; i++) {
		string eat;
		cin >> eat;
		if (eat == "Adopted") {
			break;
		}
		stringstream geek(eat);
		int x = 0;
		geek >> x;
		allFood = allFood + x;
	}
	int leftFood = abs(grToKg - allFood);
	if (grToKg >= leftFood) {
		cout << "Food is enough! Leftovers: " << leftFood << " grams.";

	}
	else if (grToKg < leftFood) {
		cout << "Food is not enough. You need " << leftFood << " grams more.";
	}
	return 0;
}
